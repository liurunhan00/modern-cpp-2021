cmake_minimum_required(VERSION 3.10)
project(input_parameters_stream)

set(CMAKE_CXX_STANDARD 17)
#  generate compiple commands file
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)
set(CMAKE_CXX_FLAGS "-Wall")

## SET FOLDER
# set headers folder
#  c++ -I
include_directories(include/)
# set libs folder
# c++ -L
link_directories()
set(LIBRARY_OUTPUT_PATH ${PROJECT_SOURCE_DIR}/results/lib)
# set executables folder
set(EXECUTABLE_OUTPUT_PATH ${PROJECT_SOURCE_DIR}/bin)

## COMPILATION
add_library(tools src/tools.cc)
add_executable(task_2 src/main.cc)
target_link_libraries(task_2 tools)
